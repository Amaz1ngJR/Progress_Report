def forward(self, x):
        x, x_downsample = self.forward_features(x) #x:[1,3,512,512]=>=>=>[1,256,768]  #x:[1,256,768] x_downsample<x>(4,x)
        x = self.forward_up_features(x,x_downsample) #x:[1,256,768] =>[1,16384,96]

        return x
